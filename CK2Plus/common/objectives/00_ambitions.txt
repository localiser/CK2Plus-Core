obj_become_chancellor = {
	type = character

	expectation_of_liege = yes

	allow = {
		NOT = { liege = { spouse = { character = FROM } } }
		prisoner = no
		is_adult = yes
		is_councillor = no
		diplomacy = 8
		liege = {
			NOT = { character = ROOT }
		}
		OR = {
			is_female = no
			liege = {
				primary_title = {
					has_law = true_cognatic_succession
				}
			}
		}
		NOT = { trait = incapable }
		NOT = { has_character_flag = obj_become_councillor }
		diplomacy = 10
	}
	chance = {
		factor = 100

		modifier = {
			factor = 0.01
			trait = content
		}
		modifier = {
			factor = 0.1
			trait = shy
		}
		modifier = {
			factor = 0.25
			trait = slothful
		}
		modifier = {
			factor = 1.25
			from_ruler_dynasty = yes
		}
		modifier = {
			factor = 1.5
			trait = poet
		}
		modifier = {
			factor = 1.5
			trait = falconer
		}
		modifier = {
			factor = 1.5
			trait = gregarious
		}
		modifier = {
			factor = 2.0
			trait = ambitious
		}
		modifier = {
			factor = 1.5
			trait = naive_appeaser
		}
		modifier = {
			factor = 2.0
			trait = underhanded_rogue
		}
		modifier = {
			factor = 10.0
			OR = {
				trait = charismatic_negotiator
				trait = grey_eminence
			}
		}
		modifier = {
			factor = 1.25
			diplomacy = 12
		}
		modifier = {
			factor = 1.25
			diplomacy = 14
		}
		modifier = {
			factor = 1.25
			diplomacy = 16
		}
		modifier = {
			factor = 1.5
			diplomacy = 18
		}
		modifier = {
			factor = 1.5
			diplomacy = 20
		}
	}
	success = {
		has_job_title = job_chancellor
	}
	abort = {
		OR = {
			prisoner = yes
			independent = yes
			AND = {
				is_councillor = yes
				NOT = { has_job_title = job_chancellor }
			}
			liege = { spouse = { character = FROM } }
		}
	}
	abort_effect = {
		set_character_flag = obj_aborted
		clr_character_flag = obj_check
	}
	effect = {
		if = {
			limit = {
				NOT = { has_character_flag = obj_become_councillor }
			}
			change_diplomacy = 1
		}
		opinion = {
			who = liege
			modifier = opinion_ambition_fulfilled
			months = 120
		}
		set_character_flag = obj_become_councillor
	}
}
