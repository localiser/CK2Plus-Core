#newborn characters
character_event = {
	id = 300002
	desc = "EVTDESC300001"

	trigger = {
		NOT = { age = 1 }
		NOT = { has_character_flag = genetics_check }
	}

		days = 1
	}

		name = "EVTOPTA300001"
		set_character_flag = genetics_check

			add_trait = young1

				limit = {
					NOT = { trait = divine_blood }

						trait = divine_blood 
					}
				}
				add_trait = divine_blood				
			}
			if = {
				limit = {
					NOT = { trait = divine_blood }

						dynasty = ROOT 
						trait = divine_blood 
					}
				}
				add_trait = divine_blood				
			}	
			if = {
				limit = {
					OR = {
						father = { trait = immortal }
						mother = { trait = immortal }
					}
				}

					chance = 10
					add_trait = immortal
				}
			}
			
			#check for stillbirth
			character_event = { id = 190000 }			

			any_child = {
				if = {
					limit = { ROOT = { is_female = no } }

						modifier = opinion_father
						who = ROOT
					}
					opinion = {
						modifier = opinion_father
						who = ROOT
						years = 100
					}					
				}	
				any_child = {
					if = {
						limit = { ROOT = { is_female = no } }

							modifier = opinion_grandfather

							}

						}
					}
					remove_opinion = {
						modifier = opinion_sibling
						who = ROOT
					}
					reverse_remove_opinion = {
						modifier = opinion_sibling
						who = ROOT
					}
					opinion = {
						modifier = opinion_sibling
						who = ROOT
						years = 100
					}
					reverse_opinion = {
						modifier = opinion_sibling
						who = ROOT
						years = 100
					}
				}
				any_child = {		#nephews/nieces
					remove_opinion = {
						modifier = opinion_uncle
						who = ROOT
					}
					remove_opinion = {
						modifier = opinion_aunt
						who = ROOT
					}
					reverse_remove_opinion = {
						modifier = opinion_nephew
						who = ROOT
					}
					reverse_remove_opinion = {
						modifier = opinion_niece
						who = ROOT
					}
					if = {
						limit = {
							ROOT = {
								is_female = no
							}
						}
						opinion = {
							modifier = opinion_uncle
							who = ROOT
							years = 100
						}
					}
					if = {
						limit = {
							ROOT = {
								is_female = yes
							}
						}
						opinion = {
							modifier = opinion_aunt
							who = ROOT
							years = 100
						}
					}
					if = {
						limit = {
							is_female = no

						reverse_opinion = {
							modifier = opinion_nephew

			}		#END siblings
			father_even_if_dead = {		#father
				reverse_remove_opinion = {

				}

						who = ROOT
					}

						modifier = opinion_grandchild
						who = ROOT
						years = 100
					}
					reverse_remove_opinion = {
						modifier = opinion_grandfather
						who = ROOT
					}					
					reverse_opinion = {
						modifier = opinion_grandfather
						who = ROOT
						years = 100
					}
				}		#END grandfather
				mother = {		#grandmother
					remove_opinion = {
						modifier = opinion_grandchild
						who = ROOT
					}				
					opinion = {
						modifier = opinion_grandchild
						who = ROOT
						years = 100

#Reduce fertility / change stats for older characters
character_event = {
	id = 300003
	picture = "GFX_evt_illness"
	
	is_triggered_only = yes
	hide_from = yes
	
	trigger = {
		ai = yes
		is_ruler = no
		liege = { 
			NOT = { trait = immortal }
			age = 50 
			NOT = { age = 60 }
			ai = no
			NOT = { trait = old50 }
		}
	}
	
	mean_time_to_happen = { days = 1 }
		
	desc = "AI_EVENT"

	option = {
		name = "OK"
		liege = {
			add_trait = old50
			remove_trait = young1
			remove_trait = young6
			remove_trait = young16			
		}
	}	
}
