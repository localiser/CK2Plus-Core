character_event = {
	id = 1100099991
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		siege = { is_attacker = yes }
		location = {
			any_province_lord = {
				siege = { is_defender = yes }
				OR = {
					any_courtier = {
						is_close_relative = PREV
						at_location = ROOT
						in_command = no
						NOT = { has_character_modifier = left_at_siege }
						NOT = {	has_character_modifier = modifier_hidden }
						NOT = { has_character_modifier = stalwart_defender }
					}
					AND = {
						at_location = ROOT
						NOT = { has_character_modifier = left_at_siege }
						NOT = {	has_character_modifier = modifier_hidden }
						NOT = { has_character_modifier = stalwart_defender }
					}
				}
			}
		}
	}
	
	immediate = {
		location = {
			any_province_lord = {
				limit = {
					siege = { is_defender = yes }
					OR = {
						any_courtier = {
							is_close_relative = PREV
							at_location = ROOT
							in_command = no
							NOT = { has_character_modifier = left_at_siege }
							NOT = {	has_character_modifier = modifier_hidden }
							NOT = { has_character_modifier = stalwart_defender }
						}
						AND = {
							at_location = ROOT
							NOT = { has_character_modifier = left_at_siege }
							NOT = {	has_character_modifier = modifier_hidden }
							NOT = { has_character_modifier = stalwart_defender }
						}
					}
				}
				if = {
					limit = {
						any_courtier = {
							is_close_relative = PREV
							at_location = ROOT
							in_command = no
							NOT = { has_character_modifier = left_at_siege }
							NOT = {	has_character_modifier = modifier_hidden }
							NOT = { has_character_modifier = stalwart_defender }
						}
					}
					set_character_flag = family_present_at_siege
				}
				if = {
					limit = {
						at_location = ROOT
						NOT = { has_character_modifier = left_at_siege }
						NOT = {	has_character_modifier = modifier_hidden }
						NOT = { has_character_modifier = stalwart_defender }
					}
					set_character_flag = personally_present_at_siege
				}
				character_event = { id = 1100099998 }
			}
		}
	}
	
	option = {
		name = OK
	}
