# Drunkard
character_event = {
	id = 91200
	desc = "EVTDESC91200"
	picture = "GFX_evt_drunk"

	min_age = 16
	capable_only = yes
	prisoner = no
	only_rulers = yes

	trigger = {
		religion_group = muslim
		OR = { higher_tier_than = baron is_patrician = yes }
		trait = drunkard
		NOT = {
			any_liege = {
				dynasty = ROOT
			}
		}
		NOT = { has_character_flag = decadence_drunkard }
	}

	mean_time_to_happen = {
		months = 300
		modifier = {
			factor = 2.0
			always = no
		}
		modifier = {
			factor = 2.0
			has_character_flag = decadence_change
			NOT = {
				had_character_flag = {
					flag = decadence_change
					days = 1825
				}
			}
		}
		modifier = {
			factor = 0.75
			tier = emperor
		}
		modifier = {
			factor = 0.9
			tier = king
		}
		modifier = {
			factor = 2.0
			tier = count
		}
	}

	option = {
		name = "EVTOPTA91200"
		decadence = 5
		custom_tooltip = {
			text = RELIGIOUSFACTIONMOODMINUS
			hidden_tooltip = { change_variable = { which = "faction_mood_religious" value = -1 } }
			hidden_tooltip = { random_courtier = { limit = { ai = yes } character_event = { id = 1000183004 days = 1 } } }
		}
		set_character_flag = decadence_drunkard
		set_character_flag = decadence_change
	}
}
